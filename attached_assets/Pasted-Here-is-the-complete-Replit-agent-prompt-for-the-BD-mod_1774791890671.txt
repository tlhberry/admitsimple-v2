3. API ROUTES
Add all of the following routes to server/routes.ts. All routes are scoped to companyId from the authenticated session. BD reps only see accounts assigned to them; admins/staff see all.
typescript// ── Referral Accounts ──────────────────────────────────────────────

// Get all referral accounts for company (bd_rep sees only their assigned accounts)
app.get("/api/referral-accounts", canAccessReferralAccounts, async (req, res) => {
  const { companyId, userId, role } = req.user;
  let accounts;
  if (role === "bd_rep") {
    accounts = await db.select().from(referralAccounts)
      .where(and(eq(referralAccounts.companyId, companyId), eq(referralAccounts.assignedBdRepId, userId)));
  } else {
    accounts = await db.select().from(referralAccounts)
      .where(eq(referralAccounts.companyId, companyId));
  }
  res.json(accounts);
});

// Create referral account
app.post("/api/referral-accounts", canAccessReferralAccounts, async (req, res) => {
  const { companyId, userId } = req.user;
  const data = insertReferralAccountSchema.parse({ ...req.body, companyId, createdBy: userId });
  const [account] = await db.insert(referralAccounts).values(data).returning();
  res.json(account);
});

// Get single referral account by id
app.get("/api/referral-accounts/:id", canAccessReferralAccounts, async (req, res) => {
  const account = await db.select().from(referralAccounts)
    .where(eq(referralAccounts.id, parseInt(req.params.id)));
  if (!account.length) return res.status(404).json({ message: "Account not found" });
  res.json(account[0]);
});

// Update referral account
app.patch("/api/referral-accounts/:id", canAccessReferralAccounts, async (req, res) => {
  const [account] = await db.update(referralAccounts)
    .set({ ...req.body, updatedAt: new Date() })
    .where(eq(referralAccounts.id, parseInt(req.params.id)))
    .returning();
  res.json(account);
});

// Delete referral account
app.delete("/api/referral-accounts/:id", canAccessReferralAccounts, async (req, res) => {
  await db.delete(referralAccounts).where(eq(referralAccounts.id, parseInt(req.params.id)));
  res.json({ success: true });
});

// Generate AI insights for a referral account using Claude
app.post("/api/ai/referral-insights", canAccessActivities, async (req, res) => {

  const [account] = await db.select().from(referralAccounts).where(eq(referralAccounts.id, accountId));

Analyze this referral account and provide actionable insights:

Account: ${account.name}

Contacts (${contacts.length}):

Please provide:

  const response = await anthropic.messages.create({
